    <script>
        document.getElementById('myForm').onsubmit = async function(e) {
            e.preventDefault();
            const res = document.getElementById('response');
            res.className = '';
            res.style.display = 'none';
            
            const data = {
                name: document.getElementById('name').value,
                email: document.getElementById('email').value,
                phone: document.getElementById('phone').value
            };

            try {
                await fetch('https://script.google.com/macros/s/AKfycbyvx07s-Pv1C5ahuWyKnfv67eHJ3cAx0PcR9rHF34am2ship3MMKMvLD2OvGUkMfH1J/exec', {
                    method: 'POST',
                    mode: 'no-cors',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify(data)
                });
                res.textContent = '✅ تم الإرسال بنجاح!';
                res.className = 'success';
                document.getElementById('myForm').reset();
            } catch(error) {
                res.textContent = '❌ خطأ: ' + error.message;
                res.className = 'error';
            }
        };
    </script>
